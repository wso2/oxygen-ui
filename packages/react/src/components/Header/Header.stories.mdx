## Overview

The `Header` provides content and actions related to the current screen. It's used for branding, screen titles,
navigation, and actions.

export const basicHeaderVariationOptions = {
  brand: {
    logo: {
      desktop: <img src="assets/images/oxygen-ui-logo-mini.svg" alt="logo" height="40" />,
      mobile: <img src="assets/images/oxygen-ui-logo-mini.svg" alt="logo" height="40" />,
    },
    onClick: () => null,
    title: 'OxygenUI',
  },
  user: {
    image: 'assets/images/avatar-john.svg',
    name: 'Matthew',
    email: 'matthew@wso2.com',
  },
  modes: [
    {
      name: 'light',
      icon: <SunIcon />,
    },
    {
      name: 'dark',
      icon: <CrescentBrightIcon />,
    },
  ],
};
